#define potPin A0
int aRead = 0;
int tempC;
void setup()
{
pinMode(7, OUTPUT);
pinMode(8, OUTPUT);
Serial.begin(9600);
}

void loop ()
{
aRead = 0;
aRead = analogRead(potPin);
tempC = aRead *0.48875;//0.09775 // 0.48875;
Serial.print("Temp = ");
Serial.println(tempC);
delay(300);
digitalWrite(8, HIGH);
digitalWrite(8, LOW);
for(int i = 0; i<tempC; i++) {
digitalWrite(7, HIGH);
delayMicroseconds(1);
digitalWrite(7, LOW);
delayMicroseconds(1);
}
}
